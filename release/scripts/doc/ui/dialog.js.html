396          var x = xpos,
397              w,
398              y = ypos,
399              h,
400              dim = self.dimension(),
401              Center = Dialog.Position.Center,
402              GoldenRatio = Dialog.Position.GoldenRatio,
403              parent = self.parent();
404          if (typeof ypos == 'undefined') {
405              ypos = Center;
406          }
